If you already have an old SML database, you need to perform an upgrade since the table names have been changed.
To do this for MySQL, use the following commands, assuming that the database is called "sml".
// Philip Helger, 2014-09-25

-- rename tables
ALTER TABLE `sml`.`sml_allowed_wildcard_schemes` RENAME TO  `sml`.`allowed_wildcard_schemes`;
ALTER TABLE `sml`.`sml_migrate` RENAME TO  `sml`.`migrate`;
ALTER TABLE `sml`.`sml_recipient_part_identifier` RENAME TO  `sml`.`recipient_part_identifier`;
ALTER TABLE `sml`.`sml_service_metadata_publisher` RENAME TO  `sml`.`service_metadata_publisher`;

-- drop FK constraints
ALTER TABLE `sml`.`service_metadata_publisher` DROP FOREIGN KEY `FK_service_metadata_publisher_1`;
ALTER TABLE `sml`.`allowed_wildcard_schemes` DROP FOREIGN KEY `new_fk_constraint`;

-- rename columns
ALTER TABLE `sml`.`sml_user` CHANGE COLUMN `username` `sml_username` VARCHAR(200) NOT NULL;
ALTER TABLE `sml`.`sml_user` CHANGE COLUMN `password` `sml_password` TEXT NULL;
ALTER TABLE `sml`.`migrate` CHANGE COLUMN `recipient_participant_identifier_scheme` `scheme` varchar(200) NOT NULL;
ALTER TABLE `sml`.`migrate` CHANGE COLUMN `recipient_participant_identifier_value` `rec_value` varchar(200) NOT NULL;
ALTER TABLE `sml`.`recipient_part_identifier` CHANGE COLUMN `recipient_participant_identifier_scheme` `scheme` varchar(200) NOT NULL;
ALTER TABLE `sml`.`recipient_part_identifier` CHANGE COLUMN `recipient_participant_identifier_value` `rec_value` varchar(200) NOT NULL;

-- recreate FK constraints
ALTER TABLE `sml`.`service_metadata_publisher` 
  ADD CONSTRAINT `FK_service_metadata_publisher_1`
  FOREIGN KEY (`username`)
  REFERENCES `sml`.`sml_user` (`sml_username`)
  ON DELETE CASCADE
  ON UPDATE CASCADE;
ALTER TABLE `sml`.`allowed_wildcard_schemes` 
  ADD CONSTRAINT `new_fk_constraint`
  FOREIGN KEY (`username`)
  REFERENCES `sml`.`sml_user` (`sml_username`)
  ON DELETE CASCADE
  ON UPDATE CASCADE;
